CREATE TABLE farmsizes AS (SELECT to_char(state_fips_code, '99')|| to_char(county_code, 'fm000') as fips, year,  value as farm_count, domain_category_description as size_category FROM raw_nass_census_data WHERE source_desc='CENSUS' AND statistical_category_desc='OPERATIONS' AND commodity_desc='FARM OPERATIONS' AND unit_desc='OPERATIONS' AND sector_desc='ECONOMICS' AND domain_category_description IN ('AREA OPERATED: (1.0 TO 9.9 ACRES)', 'AREA OPERATED: (10.0 TO 49.9 ACRES)', 'AREA OPERATED: (50 TO 179 ACRES)', 'AREA OPERATED: (180 TO 499 ACRES)', 'AREA OPERATED: (500 TO 999 ACRES)', 'AREA OPERATED: (1,000 OR MORE ACRES)') );
